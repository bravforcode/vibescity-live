    ```js
    const NEON_CATEGORY_COLORS = {
      bar: '#ff00ff',
      cocktail: '#ff00ff',
      nightclub: '#ff00ff',
      music: '#ff4444',
      live: '#ff4444',
      food: '#00ff88',
      street: '#00ff88',

      edible: '#44ff44',
      cafe: '#ffdd00',
      gallery: '#ffdd00',
      art: '#ffdd00',
      spa: '#ffdd00',
    };

    const getNeonColor = (category) => {
      const lower = String(category || '').toLowerCase();
      for (const [key, color] of Object.entries(NEON_CATEGORY_COLORS)) {
        if (lower.includes(key)) return color;
      }
      return '#00e5ff';
    };
    ```

    2. Replace the body of the existing `export const createMarkerElement` function with the neon sign implementation. Keep the export name identical — useMapMarkers.js imports it by this name and must not need changes.

    The new function body (all user strings are escaped with escapeHtml before use in innerHTML):
    ```js
    export const createMarkerElement = ({ item, isHighlighted, isLive, hasCoins = true }) => {
      const el = document.createElement('div');
      const neonColor = getNeonColor(item.category || item.type || '');
      el.className = 'neon-sign-marker' + (isHighlighted ? ' neon-sign-selected' : '');
      el.style.setProperty('--neon-color', neonColor);
      el.dataset.shopId = sanitizeId(item.id);

      // escapeHtml applied to all user-supplied strings before innerHTML insertion
      const safeName = escapeHtml(item.name || 'VIBE');
      const truncated = safeName.length > 14 ? safeName.substring(0, 14) : safeName;

      el.innerHTML =
        (hasCoins ? '<div class="neon-coin-float lottie-coin-target"></div>' : '') +
        '<span class="neon-sign-text">' + truncated + '</span>' +
        (isLive ? '<span class="neon-sign-badge">LIVE</span>' : '');

      return el;
    };
    ```
